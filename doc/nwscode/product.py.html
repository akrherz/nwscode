0019class ProductError(Exception):
0020    pass
0021
0022class Product (object):
0023    def __init__(self, text):
0024        self.text = text.strip().replace("\r\n", "\n")
0025        m = Ugc.pattern.search(text)
0026        if not m:
0027            raise ProductError("Product does not contain a UGC code.")
0028        self.header = Header(self.text[:m.start()].strip())
0029        body = self.text[m.start():].strip()
0030        self.segments = []
0031        segment_texts = Segment.pattern.split(body)
0032        self.footer = Footer(segment_texts.pop())
0033        for seg_text in segment_texts:
0034            events = []
0035            seg_text = seg_text.strip()
0036            if not seg_text:
0037                continue
0038            try:
0039                seg = Segment(seg_text)
0040            except SegmentError:
0041                pass
0042            else:
0043                self.segments.append(seg)
0044
0045    def __str__(self):
0046        lines = [str(self.header)]
0047        if self.segments:
0048            lines.extend([str(ev) for ev in self.segments])
0049        lines.append(str(self.footer))
0050        return '\n'.join(lines)

0055    ``Product Header``

0061            self.wmo = WmoHeader(text[0])
0062            self.awipsid = AwipsId(text[1])
0063        except:
0064            raise ProductError()
0065
0066    def __str__(self):
0067        return self.text

0077class SegmentError(ProductError):
0078    pass
0079
0080SEGMENT = r"^\$\$$"
0081class Segment(object):
0082    """ Segment wraps a text segment.
0083        
0084        Attribute:
0085        
0086            ``ugc``
0087                ugc code for this segment.
0088        
0089            ``text``
0090                the actual segment text.
0091        
0092            ``events``
0093                a list of any events in the segment.
0094        
0095            ``forecasts``
0096                list of forecasts in the segment.  Ordered nearest to farthest
0097                from now.
0098            
0099            ``headlines``
0100                list of headlines in the segment.
0101        
0102    """
0103    pattern = re.compile(SEGMENT, re.MULTILINE|re.DOTALL)
0104    def __init__(self, text):
0105        self.text = text
0106        # parse events.
0107        self.events = []
0108        m = Ugc.pattern.search(self.text)
0109        if not m:
0110            raise SegmentError("Segment does not have a UGC code.")
0111        self.ugc = Ugc(self.text[m.start():m.end()])
0112        for line in self.text.split('\n'):
0113            if Pvtec.pattern.match(line):
0114                self.events.append(Event(self.ugc, Pvtec(line)))
0115            elif Hvtec.pattern.match(line):
0116                self.events[-1].hvtec = Hvtec(line)
0117        # parse forecasts.
0118        self.forecasts = []
0119        fcst_pat = re.compile(r"(?m)^\.[A-Z ]+?[\.]{3}.*(?:\n(?:[A-Z0-9].*)*)*")
0120        for m in fcst_pat.finditer(self.text):
0121            fcst = m.group(0).replace("\n", " ").strip()
0122            self.forecasts.append(fcst)
0123        # parse headlines.
0124        self.headlines = []
0125        headline_pat = re.compile(r"(?ms)^[\.]{3}[A-Z0-9 \n]*[\.]{3}$")
0126        for m in headline_pat.finditer(self.text):
0127            headline = m.group(0).replace("\n", " ").strip()
0128            self.headlines.append(headline)
0129        # we've got to handle short-fuse products seperately.  These don't
0130        # even have a headline in them so we have to generate them.
0131        short_fuse = re.compile(r"(?m)^\* (?:SEVERE THUNDERSTORM|TORNADO|(?:FLASH )?FLOOD) (?:WARNING|WATCH|ADVISORY) FOR[\.]*(?:\n[A-Z0-9 .]+)$")
0132        for m in short_fuse.finditer(self.text):
0133            headline = m.group(0)
0134            headline = headline.replace('\n', ' ')
0135            headline = headline.lstrip("* ")
0136            headline = headline.replace("...", "")
0137            headline = re.compile("  +").sub(' ', headline)
0138            headline = "..." + headline + "..."
0139            self.headlines.append(headline)
0140
0141    def __str__(self):
0142        return self.text

0145class Event(object):
0146    """
0147    ``Event`` wraps instances of a UGC, a PVTEC, and (optionally) an HVTEC
0148    object.  The attributes of each of those objects are accessible through
0149    the Event object.  The event object is intended as a minimal container
0150    for these three objects.
0151    """
0152
0153    def __init__(self, ugc, pvtec, hvtec=None):
0154        self.ugc = ugc
0155        self.pvtec = pvtec
0156        self.hvtec = hvtec
0157
0158    def __getattr__(self, name):
0159        if hasattr(self.ugc, name):
0160            return getattr(self.ugc, name)
0161        elif hasattr(self.pvtec, name):
0162            return getattr(self.pvtec, name)
0163        if self.hvtec and hasattr(self.hvtec, name):
0164            return getattr(self.hvtec, name)
0165        raise AttributeError
0166
0167    def __str__(self):
0168        s = [str(self.ugc), str(self.pvtec)]
0169        if self.hvtec:
0170            s.append(str(self.hvtec))
0171        return '\n'.join(s)
0172
0173    def __repr__(self):
0174        return repr(str(self))
