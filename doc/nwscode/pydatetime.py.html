0080_DI400Y = _days_before_year(401)    # number of days in 400 years

0096def _ord2ymd(n):
0097    "ordinal -> (year, month, day), considering 01-Jan-0001 as day 1."
0098
0099    # n is a 1-based index, starting at 1-Jan-1.  The pattern of leap years
0100    # repeats exactly every 400 years.  The basic strategy is to find the
0101    # closest 400-year boundary at or before n, then work with the offset
0102    # from that boundary to n.  Life is much clearer if we subtract 1 from
0103    # n first -- then the values of n at 400-year boundaries are exactly
0104    # those divisible by _DI400Y:
0105    #
0106    #     D  M   Y            n              n-1
0107    #     -- --- ----        ----------     ----------------
0108    #     31 Dec -400        -_DI400Y       -_DI400Y -1
0109    #      1 Jan -399         -_DI400Y +1   -_DI400Y      400-year boundary
0110    #     ...
0111    #     30 Dec  000        -1             -2
0112    #     31 Dec  000         0             -1
0113    #      1 Jan  001         1              0            400-year boundary
0114    #      2 Jan  001         2              1
0115    #      3 Jan  001         3              2
0116    #     ...
0117    #     31 Dec  400         _DI400Y        _DI400Y -1
0118    #      1 Jan  401         _DI400Y +1     _DI400Y      400-year boundary
0119    n -= 1
0120    n400, n = divmod(n, _DI400Y)
0121    year = n400 * 400 + 1   # ..., -399, 1, 401, ...
0122
0123    # Now n is the (non-negative) offset, in days, from January 1 of year, to
0124    # the desired date.  Now compute how many 100-year cycles precede n.
0125    # Note that it's possible for n100 to equal 4!  In that case 4 full
0126    # 100-year cycles precede the desired day, which implies the desired
0127    # day is December 31 at the end of a 400-year cycle.
0128    n100, n = divmod(n, _DI100Y)
0129
0130    # Now compute how many 4-year cycles precede it.
0131    n4, n = divmod(n, _DI4Y)
0132
0133    # And now how many single years.  Again n1 can be 4, and again meaning
0134    # that the desired day is December 31 at the end of the 4-year cycle.
0135    n1, n = divmod(n, 365)
0136
0137    year += n100 * 100 + n4 * 4 + n1
0138    if n1 == 4 or n100 == 4:
0139        assert n == 0
0140        return year-1, 12, 31
0141
0142    # Now the year is correct, and n is the offset from January 1.  We find
0143    # the month via an estimate that's either exact or one too large.
0144    leapyear = n1 == 3 and (n4 != 24 or n100 == 3)
0145    assert leapyear == _is_leap(year)
0146    month = (n + 50) >> 5
0147    preceding = _DAYS_BEFORE_MONTH[month] + (month > 2 and leapyear)
0148    if preceding > n:  # estimate is too large
0149        month -= 1
0150        preceding -= _DAYS_IN_MONTH[month] + (month == 2 and leapyear)
0151    n -= preceding
0152    assert 0 <= n < _days_in_month(year, month)
0153
0154    # Now the year and month are correct, and n is the offset from the
0155    # start of that month:  we're done!
0156    return year, month, n+1

0292# Notes on comparison:  In general, datetime module comparison operators raise
0293# TypeError when they don't know how to do a comparison themself.  If they
0294# returned NotImplemented instead, comparison could (silently) fall back to
0295# the default compare-objects-by-comparing-their-memory-addresses strategy,
0296# and that's not helpful.  There are two exceptions:
0297#
0298# 1. For date and datetime, if the other object has a "timetuple" attr,
0299#    NotImplemented is returned.  This is a hook to allow other kinds of
0300#    datetime-like objects a chance to intercept the comparison.
0301#
0302# 2. Else __eq__ and __ne__ return False and True, respectively.  This is
0303#    so opertaions like
0304#
0305#        x == y
0306#        x != y
0307#        x in sequence

0309#        dict[x] = y
0310#
0311#    don't raise annoying TypeErrors just because a datetime object
0312#    is part of a heterogeneous collection.  If there's no known way to
0313#    compare X to a datetime, saying they're not equal is reasonable.
