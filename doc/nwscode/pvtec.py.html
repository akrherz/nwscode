0016# PVTEC Grammar.
0017FIXEDIDENT = r"[OTEX]"
0018ACTION = r"[A-Z]{3}"
0019OFFICEIDENT = r"[A-Z]{4}"
0020PHENOMENA = r"[A-Z]{2}"
0021SIGNIFICANCE = r"[WAYSFON]"
0022ETN = r"[0-9]{4}"
0023EVENTBEGIN = r"[0-9]{6}T[0-9]{4}Z"
0024EVENTEND = EVENTBEGIN
0025elements = (FIXEDIDENT, ACTION, OFFICEIDENT, PHENOMENA,
0026            SIGNIFICANCE, ETN, EVENTBEGIN, EVENTEND)
0027PVTEC = r"^/(%s)\.(%s)\.(%s)\.(%s)\.(%s)\.(%s)\.(%s)\-(%s)/$" % elements
0028
0029class PvtecError(NwsCodeError):
0030    pass
0031
0032class Pvtec(NwsCode):
0033    """
0034    A ``Primary Valid Time Event NwsCode`` parser.
0035    
0036    For a full description of the P-VTEC format see the National Weather
0037    Service's VTEC Directive:
0038        
0039        http://www.nws.noaa.gov/directives/sym/pd01017003curr.pdf
0040    
0041    Attributes:
0042        
0043        ``fixedid``
0044            The Fixed Identifier. Identifies the following product and VTEC
0045            code string types.
0046        
0047        ``action``
0048            Identifies the action in the product issuance.
0049        
0050        ``officeid``
0051            The standard four-letter identifier indicating the NWS office
0052            with the primary responsibility for the affected area.
0053        
0054        ``phenomena``
0055            Identifies the type of weather, flood, marine, fire weather,
0056            etc., occurrence (e.g., freezing rain, river flood, gale, red
0057            flag), or non-weather occurrence (e.g., ashfall).
0058        
0059        ``significance``
0060            Identifies the level of importance (e.g., watch, warning,
0061            advisory, etc.) of the weather or non-weather occurrence.
0062        
0063        ``etn``
0064            The ETN is a four-digit number assigned to keep track of how an
0065            event is addressed by various VTEC actions and/or products issued
0066            over the lifetime of the event.
0067        
0068        ``eventbegin``
0069            Start of the valid time span for the event.
0070        
0071        ``eventend``
0072            End of the valid time span for the event.
0073    
0074    Usage Example:
0075    
0076    >>> from nwscode.pvtec import PVTEC
0077    >>> pvtec_string = '/X.EXT.PAFG.FG.Y.0002.000000T0000Z-060127T2100Z/'
0078    >>> pv = PVTEC(pvtec_string)
0079    >>> pv.action
0080    'Extended in time'
0081    >>> pv.significance
0082    'Advisory'
0083    >>>
0084    """

0172    def _process_matches(self, matches):
0173        self.code = Bunch(fixedid=matches[0],
0174                        action=matches[1],
0175                        officeid=matches[2],
0176                        phenomena=matches[3],
0177                        significance=matches[4],
0178                        etn=matches[5],
0179                        eventbegin=matches[6],
0180                        eventend=matches[7])
0181        self.fixedid = self._interpret("fixedid", matches[0])
0182        self.action = self._interpret("action", matches[1])
0183        self.officeid = matches[2]
0184        self.phenomena = self._interpret("phenomena", matches[3])
0185        self.significance = self._interpret("significance", matches[4])
